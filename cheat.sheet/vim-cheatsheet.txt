  w / W            next word / WORD
  b / B            previous word / WORD
  e / E            end of word / WORD
  gg               top of file
  G                bottom of file
  {number}G        go to line number
  %                jump to matching bracket

LINE NAVIGATION  (remapped for ergonomics)
  H                first non-blank character  (remapped from ^)
  L                end of line                (remapped from $)
  0                absolute start of line

CHARACTER JUMPS
  f{char}          jump forward to char (inclusive)
  F{char}          jump backward to char
  t{char}          jump before char (exclusive)
  T{char}          jump before char (backward)
  ;                repeat last f/t motion
  ,                repeat in opposite direction

COUNTS
  5j               move 5 lines down
  3w               jump 3 words forward

  x                delete character under cursor
  dd               delete line
  D                delete to end of line
  yy               yank (copy) line
  p / P            paste after / before cursor
  r{char}          replace single character
  u                undo
  Ctrl-r           redo
  .                repeat last change   ← use this constantly
  *                search word under cursor (forward)
  #                search word under cursor (backward)
  Ctrl-a           increment number under cursor
  Ctrl-x           decrement number under cursor

  d + motion       delete
  c + motion       change (delete + enter insert)
  y + motion       yank

  dw               delete word
  cw               change word
  dL               delete to end of line    (remapped L = $)
  cH               change to first non-blank (remapped H = ^)
  df,              delete until comma (inclusive)
  ct)              change until )
  cc               change whole line
  >>               indent line
  <<               unindent line
  =ap              auto-format paragraph

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  i = inner (excludes surrounding)
  a = around (includes surrounding)

  ciw / diw / yiw      change / delete / yank inner word
  caw / daw            change / delete around word (includes space)

  ci" / di"            inside double quotes
  ci' / di'            inside single quotes
  ci` / di`            inside backticks

  ci( / di(            inside ()
  ci{ / di{            inside {}
  ci[ / di[            inside []
  da{ / da(            delete including the brackets

  vip                  select inner paragraph
  vap                  select around paragraph

  a                append after cursor

  O                new line above and enter insert

  Esc / Ctrl-[     return to Normal

SHORTCUTS IN INSERT
  Ctrl-w           delete previous word
  Ctrl-u           delete to start of line
  Ctrl-h           delete character (backspace)
  Ctrl-n / Ctrl-p  next / previous completion

  V                line-wise
  Ctrl-v           block mode
  o                switch active selection edge

ACTIONS
  y                yank selection
  d                delete selection
  c                change selection
  > / <            indent / unindent (stays in visual)
  =                auto-indent
  p                paste (won't overwrite register — remapped)

MOVE LINES (remapped)
  J                move selection down
  K                move selection up

BLOCK EDITING
  Ctrl-v + j + I   insert at start of multiple lines
  Ctrl-v + j + A   append at end of multiple lines

  ?pattern         search backward
  n                next match       (centers screen — remapped)
  N                previous match   (centers screen — remapped)
  *                search word under cursor
  #                search backward word under cursor
  <leader>nh       clear search highlight

    lowercase  →  case-insensitive
    uppercase  →  case-sensitive

  :q               quit
  :wq / :x         save and quit
  :q!              quit without saving
  :e filename      open file
  :saveas fname    save as

SEARCH & REPLACE
  :s/old/new/      replace first on line
  :s/old/new/g     replace all on line
  :%s/old/new/g    replace entire file
  :%s/old/new/gc   confirm each replace
  :g/pattern/d     delete all matching lines

SPLITS
  :split           horizontal split
  :vsplit          vertical split
  Ctrl-w + s       split horizontal
  Ctrl-w + v       split vertical
  Ctrl-w h/j/k/l   move between splits   (also <leader>h/j/k/l)
  Ctrl-w =         equalize splits
  Ctrl-w q         close split

TABS
  <leader>t        new tab
  Tab              next tab      (remapped)
  Shift-Tab        previous tab  (remapped)
  gt / gT          next / previous tab
  :tabclose        close tab

11. LEADER KEY MAPPINGS  (Leader = Space)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  <leader>w        save file
  <leader>q        quit
  <leader>x        save and quit
  <leader>t        new tab
  <leader>nh       clear search highlight
  <leader>o        blank line below (stay in Normal)
  <leader>O        blank line above (stay in Normal)
  <leader>d        delete to black hole register (won't overwrite yank)

  Window navigation:
  <leader>h/j/k/l  move between splits

12. MACROS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  qa               start recording macro into register a
  q                stop recording
  @a               run macro a
  @@               repeat last macro
  10@a             run macro 10 times

  Tips:
  - Plan the full action before recording
  - End macro on a position ready for the next repeat
  - Combine with text objects for reliable edits
  - Use . inside macros to repeat sub-actions

  "ap              paste from register a
  "_d              delete to black hole (nothing saved)

15. ERGONOMIC REMAPS SUMMARY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  jk               Esc (insert mode)
  H                ^ first non-blank  (normal, visual, operator)
  L                $ end of line      (normal, visual, operator)
  n / N            next/prev match + center screen
  Ctrl-d / Ctrl-u  half page + center screen
  J / K            move selection up/down (visual only)
  > / <            indent and stay in visual
  p                paste without losing register (visual)
  <leader>d        delete to black hole register
  Tab / S-Tab      next / previous tab

16. PRO HABITS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  • Remap CapsLock → Esc at OS level
  • Use f/t/;/, for horizontal movement instead of holding l
  • Use text objects (ciw, ci") instead of multiple motions
  • Use . constantly — it's your most powerful key
  • Use * to search word under cursor instantly
  • Use Ctrl-o / Ctrl-i to jump between edit positions
  • Use marks (ma / 'a) for long-distance jumps in a file
  • Delete with "_d when you don't want to overwrite your yank
  • Paste in visual with p — it won't clobber your clipboard
